# Comandos Uteis - Open Claw

# Conectar no servidor (com tunnel incluso)
ssh openclaw-server

# Matar tunnel antigo antes de conectar (se der erro de porta ocupada)
lsof -ti :18789 | xargs kill -9 2>/dev/null; ssh openclaw-server

# Deploy (enviar codigo para o servidor)
cd ~/Documentos/Projetos/config-open-claw && bash scripts/deploy.sh

# Tunnel manual (alternativa ao ssh config)
bash scripts/ssh-tunnel.sh

# Tunnel como servico permanente
bash scripts/ssh-tunnel-service.sh

# Testar se o gateway responde
curl -s http://127.0.0.1:18789/health

# Verificar status do gateway (no servidor)
systemctl --user status openclaw-gateway

# Listar agentes registrados (no servidor)
OPENCLAW="$(find "$HOME/.nvm/versions/node" -maxdepth 4 -path '*/bin/openclaw' 2>/dev/null | sort -V | tail -n1)" && $OPENCLAW agents list

# Ver logs do gateway (no servidor)
journalctl --user -u openclaw-gateway -f
